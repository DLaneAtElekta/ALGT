* CDB breakpoint script for AutomataLib.dll tracing
* Wait for AutomataLib.dll to load, then set breakpoints
sxe ld:AutomataLib
g
.echo *** CDB: AutomataLib.dll loaded, setting breakpoints ***
bu AutomataLib!CAInit ".echo TRACE_ENTER CAInit; gu; .echo TRACE_EXIT CAInit eax=; r eax; gc"
